   ```typescript
   import { openai } from "../lib/openai.js";
   import { prisma } from "../lib/prisma.js";
   import { logger } from "../lib/logger.js";

   export interface TrendingKeyword {
     keyword: string;
     searchVolume: number;
     trendVelocity: number; // % growth
     region: string;
     relatedKeywords: string[];
   }

   export class TrendAgent {
     async discoverTrends({
       niche,
       region = 'US',
       limit = 10,
     }: {
       niche: string;
       region?: string;
       limit?: number;
     }): Promise<TrendingKeyword[]> {
       logger.info(`[TrendAgent] Discovering trends for niche: ${niche}, region: ${region}`);

       // Note: Google Trends API requires special access
       // For MVP: Use GPT-4 to suggest trending topics
       const prompt = `List ${limit} trending topics and keywords in the ${niche} industry for ${region} market. Include estimated search volume and growth trend.

Return JSON array:
[
  {
    "keyword": "string",
    "searchVolume": number (monthly),
    "trendVelocity": number (% growth),
    "region": "${region}",
    "relatedKeywords": ["array", "of", "related", "terms"]
  }
]`;

       const response = await openai.chat.completions.create({
         model: "gpt-4o-mini",
         messages: [{ role: "user", content: prompt }],
         response_format: { type: "json_object" },
         temperature: 0.7,
       });

       const data = JSON.parse(response.choices[0]?.message?.content ?? "{}");
       return Array.isArray(data.trends) ? data.trends : [];
     }

     async subscribeToTrends({
       userId,
       keywords,
       threshold = 50,
     }: {
       userId: string;
       keywords: string[];
       threshold?: number;
     }): Promise<string> {
       logger.info(`[TrendAgent] Creating trend subscription for user: ${userId}`);

       // Create subscription record in database
       // (Use existing table or create TrendSubscription model)
       const subscription = await prisma.objective.create({
         data: {
           userId,
           organizationId: "", // Get from user context
           title: `Trend Alert: ${keywords.join(', ')}`,
           description: `Alert when keywords exceed ${threshold}% growth`,
           metadata: {
             type: "trend_subscription",
             keywords,
             threshold,
           },
           status: "active",
         },
       });

       return subscription.id;
     }

     async checkSubscriptions(): Promise<void> {
       // Daily job: Check all active subscriptions
       // Compare current trends vs. baseline
       // Send alerts if threshold exceeded
       logger.info("[TrendAgent] Checking trend subscriptions");
     }
   }

   export const trendAgent = new TrendAgent();
   ```

2. Add tRPC router: apps/api/src/trpc/routers/trends.router.ts
   ```typescript
   import { z } from "zod";
   import { router, protectedProcedure } from "../trpc.js";
   import { trendAgent } from "../../agents/TrendAgent.js";

   export const trendsRouter = router({
     discover: protectedProcedure
       .input(z.object({
         niche: z.string().min(3).max(100),
         region: z.string().length(2).default('US'),
         limit: z.number().int().min(1).max(20).default(10),
       }))
       .query(async ({ input }) => {
         return trendAgent.discoverTrends(input);
       }),

     subscribe: protectedProcedure
       .input(z.object({
         keywords: z.array(z.string().min(2)).min(1).max(20),
         threshold: z.number().int().min(10).max(200).default(50),
       }))
       .mutation(async ({ input, ctx }) => {
         return trendAgent.subscribeToTrends({
           userId: ctx.user.id,
           keywords: input.keywords,
           threshold: input.threshold,
         });
       }),

     listSubscriptions: protectedProcedure
       .query(async ({ ctx }) => {
         // Return user's trend subscriptions
         const subs = await ctx.prisma.objective.findMany({
           where: {
             userId: ctx.user.id,
             metadata: { path: ['type'], equals: 'trend_subscription' },
           },
         });
         return subs;
       }),
   });
   ```
