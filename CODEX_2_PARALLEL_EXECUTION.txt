'use client';

import { useState } from 'react';
import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Loader2 } from 'lucide-react';

export function KeywordDiscoveryPanel({ personaId }: { personaId?: number }) {
  const [seedKeywords, setSeedKeywords] = useState('');
  
  const { data, isLoading, refetch } = trpc.seo.discoverOpportunities.useQuery(
    {
      personaId,
      limit: 20,
      includeSeeds: seedKeywords.split(',').map(k => k.trim()).filter(Boolean),
    },
    { enabled: false }
  );

  return (
    <Card>
      <CardHeader>
        <CardTitle>Keyword Research</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="flex gap-2">
          <Input 
            placeholder="Enter seed keywords (e.g., ai marketing, automation)"
            value={seedKeywords}
            onChange={(e) => setSeedKeywords(e.target.value)}
            className="flex-1"
          />
          <Button 
            onClick={() => refetch()}
            disabled={!seedKeywords.trim() || isLoading}
          >
            {isLoading && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
            {isLoading ? 'Analyzing...' : 'Discover'}
          </Button>
        </div>

        {data?.opportunities && (
          <div className="space-y-2">
            <p className="text-sm text-muted-foreground">
              Found {data.opportunities.length} keyword opportunities
            </p>
            <div className="grid gap-2">
              {data.opportunities.map((opp, idx) => (
                <div key={idx} className="p-3 border rounded hover:border-primary transition-colors">
                  <div className="flex items-start justify-between">
                    <div className="flex-1">
                      <div className="font-medium">{opp.keyword}</div>
                      <div className="flex gap-2 mt-2">
                        <Badge variant="outline" className="text-xs">

                      Generate Content
                    </Button>
                  </div>
                </div>

'use client';

import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Link2, Copy, ExternalLink } from 'lucide-react';
import { useToast } from '@/hooks/use-toast';

export function InternalLinkSuggestions({ 
  contentId, 
  content 
}: { 
  contentId: string; 
  content: string; 
}) {
  const { toast } = useToast();
  const { data, isLoading } = trpc.content.suggestInternalLinks.useQuery({
    contentId,
    content: content.slice(0, 2000),
    limit: 5,
  });

  const handleCopy = (text: string) => {
    navigator.clipboard.writeText(text);
    toast({ title: 'Copied!', description: 'Anchor text copied to clipboard' });
  };

  if (isLoading) {
    return <div className="text-sm text-muted-foreground">Analyzing content for link opportunities...</div>;
  }

  if (!data || data.length === 0) {
    return (
      <Card>
        <CardContent className="py-6">
          <p className="text-sm text-muted-foreground text-center">
            No internal link suggestions found. Try adding more content.
          </p>
        </CardContent>
      </Card>
    );
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Link2 className="h-5 w-5" />
          Suggested Internal Links
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-3">
        {data.map((suggestion, idx) => (
          <div key={idx} className="p-3 border rounded hover:bg-accent transition-colors group">
            <div className="flex items-start justify-between gap-3">
              <div className="flex-1 min-w-0">
                <div className="font-medium truncate">{suggestion.title || 'Untitled Article'}</div>
                <div className="text-sm text-muted-foreground mt-1 flex items-center gap-2">
                  <span>Anchor:</span>
                  <code className="bg-muted px-2 py-0.5 rounded text-xs">
                    {suggestion.anchorText}
                  </code>
                </div>
                <Badge variant="outline" className="mt-2 text-xs">
                  Relevance: {(suggestion.similarity * 100).toFixed(0)}%
                </Badge>
              </div>
              <div className="flex gap-1">
                <Button 
                  size="sm" 
                  variant="ghost"
                  onClick={() => handleCopy(suggestion.anchorText)}
                  title="Copy anchor text"
                >
                  <Copy className="h-4 w-4" />
                </Button>
                <Button size="sm" variant="outline" title="Insert link">
                  <ExternalLink className="h-4 w-4" />
                </Button>
              </div>
            </div>
          </div>
        ))}
      </CardContent>
    </Card>
  );
}
