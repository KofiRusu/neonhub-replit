'use client';

import { useState } from 'react';
import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';

export function KeywordDiscoveryPanel({ personaId }: { personaId?: number }) {
  const [seedKeywords, setSeedKeywords] = useState('');
  const [isDiscovering, setIsDiscovering] = useState(false);
  
  const { data, isLoading, refetch } = trpc.seo.discoverOpportunities.useQuery(
    {
      personaId,
      limit: 20,
      includeSeeds: seedKeywords.split(',').map(k => k.trim()).filter(Boolean),
    },
    { enabled: false }
  );

  const handleDiscover = async () => {
    setIsDiscovering(true);
    await refetch();
    setIsDiscovering(false);
  };

  return (
    <div className="space-y-4">
      <Card>
        <CardHeader>
          <CardTitle>Keyword Research</CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          <div className="flex gap-2">
            <Input 
              placeholder="Enter seed keywords (comma-separated)"
              value={seedKeywords}
              onChange={(e) => setSeedKeywords(e.target.value)}
              className="flex-1"
            />
            <Button 
              onClick={handleDiscover}
              disabled={!seedKeywords.trim() || isDiscovering || isLoading}
            >
              {isDiscovering || isLoading ? 'Analyzing...' : 'Discover Keywords'}
            </Button>
          </div>

          {data?.opportunities && data.opportunities.length > 0 && (
            <div className="space-y-2">
              <h4 className="font-semibold text-sm">Keyword Opportunities ({data.opportunities.length})</h4>
              <div className="grid gap-2">
                {data.opportunities.map((opp, idx) => (
                  <div key={idx} className="p-3 border rounded hover:border-primary transition-colors">
                    <div className="flex items-start justify-between">
                      <div className="flex-1">
                        <div className="font-medium">{opp.keyword}</div>
                        <div className="flex gap-2 mt-1">
                          <Badge variant="outline">Difficulty: {opp.difficulty}/100</Badge>
                          <Badge variant="outline">Opportunity: {opp.opportunity}/100</Badge>
                          <Badge variant={opp.intent === 'commercial' ? 'default' : 'secondary'}>
                            {opp.intent}
                          </Badge>
                        </div>
                      </div>
                      <Button size="sm" variant="ghost">
                        Generate Content
                      </Button>
                    </div>
                  </div>
                ))}
              </div>
            </div>
          )}
        </CardContent>
      </Card>
    </div>
  );
}

'use client';

import { trpc } from '@/lib/trpc';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { TrendingUp, MousePointerClick, Eye, Target } from 'lucide-react';

export function SEODashboard({ organizationId }: { organizationId: string }) {
  const dateRange = {
    start: new Date(Date.now() - 30 * 24 * 60 * 60 * 1000).toISOString(),
    end: new Date().toISOString(),
  };

  const { data: metrics, isLoading } = trpc.seo.getMetrics.useQuery({
    organizationId,
    dateRange,
  });

  const { data: trends } = trpc.seo.getTrends.useQuery({
    organizationId,
    dateRange,
  });

  // Calculate totals
  const totals = metrics?.reduce(
    (acc, m) => ({
      impressions: acc.impressions + m.impressions,
      clicks: acc.clicks + m.clicks,
      ctr: 0,
      avgPosition: 0,
    }),
    { impressions: 0, clicks: 0, ctr: 0, avgPosition: 0 }
  ) || { impressions: 0, clicks: 0, ctr: 0, avgPosition: 0 };

  totals.ctr = totals.impressions > 0 ? (totals.clicks / totals.impressions) * 100 : 0;

  if (isLoading) {
    return <div>Loading analytics...</div>;
  }

  return (
    <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-4">
      <Card>
        <CardHeader className="flex flex-row items-center justify-between pb-2">
          <CardTitle className="text-sm font-medium">Total Impressions</CardTitle>
          <Eye className="h-4 w-4 text-muted-foreground" />
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">{totals.impressions.toLocaleString()}</div>
          <p className="text-xs text-muted-foreground">Last 30 days</p>
        </CardContent>
      </Card>

      <Card>
        <CardHeader className="flex flex-row items-center justify-between pb-2">
          <CardTitle className="text-sm font-medium">Total Clicks</CardTitle>
          <MousePointerClick className="h-4 w-4 text-muted-foreground" />
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">{totals.clicks.toLocaleString()}</div>
          <p className="text-xs text-muted-foreground">Last 30 days</p>
        </CardContent>
      </Card>

      <Card>
        <CardHeader className="flex flex-row items-center justify-between pb-2">
          <CardTitle className="text-sm font-medium">Average CTR</CardTitle>
          <Target className="h-4 w-4 text-muted-foreground" />
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold">{totals.ctr.toFixed(2)}%</div>
          <p className="text-xs text-muted-foreground">Click-through rate</p>
        </CardContent>
      </Card>

      <Card>
        <CardHeader className="flex flex-row items-center justify-between pb-2">
          <CardTitle className="text-sm font-medium">Trend</CardTitle>
          <TrendingUp className="h-4 w-4 text-muted-foreground" />
        </CardHeader>
        <CardContent>
          <div className="text-2xl font-bold text-green-600">
            {trends?.trend === 'upward' ? 'â†‘' : trends?.trend === 'downward' ? 'â†“' : 'â†’'}

          </div>
          <p className="text-xs text-muted-foreground">vs previous period</p>
        </CardContent>
      </Card>
    </div>
  );
}

'use client';

import { trpc } from '@/lib/trpc';
import { Button } from '@/components/ui/button';
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Link2, Copy } from 'lucide-react';
import { useToast } from '@/hooks/use-toast';

export function InternalLinkSuggestions({ 
  contentId, 
  content 
}: { 
  contentId: string; 
  content: string; 
}) {
  const { toast } = useToast();
  const { data, isLoading } = trpc.content.suggestInternalLinks.useQuery({
    contentId,
    content: content.slice(0, 2000), // Limit for performance
    limit: 5,
  });

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
    toast({ title: 'Copied to clipboard', description: 'Anchor text copied' });
  };

  if (isLoading) {
    return <div className="text-sm text-muted-foreground">Finding relevant links...</div>;
  }

  if (!data || data.length === 0) {
    return <div className="text-sm text-muted-foreground">No suggestions found</div>;
  }

  return (
    <Card>
      <CardHeader>
        <CardTitle className="flex items-center gap-2">
          <Link2 className="h-5 w-5" />
          Internal Link Suggestions
        </CardTitle>
      </CardHeader>
      <CardContent className="space-y-3">
        {data.map((suggestion, idx) => (
          <div 
            key={idx} 
            className="p-3 border rounded hover:bg-accent transition-colors cursor-pointer"
          >
            <div className="flex items-start justify-between">
              <div className="flex-1">
                <div className="font-medium">{suggestion.title || 'Untitled'}</div>
                <div className="text-sm text-muted-foreground mt-1">
                  Anchor: <code className="bg-muted px-1 rounded">"{suggestion.anchorText}"</code>
                </div>
                <div className="flex gap-2 mt-2">
                  <Badge variant="outline">
                    Relevance: {(suggestion.similarity * 100).toFixed(0)}%
                  </Badge>
                </div>
              </div>
              <div className="flex gap-1">
                <Button 
                  size="sm" 
                  variant="ghost"
                  onClick={() => copyToClipboard(suggestion.anchorText)}
                >
                  <Copy className="h-4 w-4" />
                </Button>
                <Button size="sm">Insert</Button>
              </div>
            </div>
          </div>
        ))}
      </CardContent>
    </Card>
  );
}
